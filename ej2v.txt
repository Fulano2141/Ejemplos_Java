 public static int contar_cir(circular cab)
 {
	 int c=0;
	 circular cab2=cab.sgte;
	 do {
		 cab2=cab2.sgte;
		 c++;
	 }while(cab2!=cab);
	 return(c);
 }

 public static circular add_nodos_circular(circular x)throws IOException
 {

  }

 }
