 public static void juego(persona d) throws IOException
 {
	 circular j= null,j2=null;

 public static int contar_cir(circular cab)
 {
	 int c=0;
	 if(cab!=null) {
	 circular cab2=cab.sgte;
	 do {
		 cab2=cab2.sgte;
		 c++;
	 }while(cab2!=cab);
	 }
	 return(c);
 }
